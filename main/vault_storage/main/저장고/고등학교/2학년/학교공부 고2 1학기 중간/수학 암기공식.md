$$
\tan\theta = \frac{\sin\theta}{\cos\theta}
$$

$$
\sin^2 \theta + \cos^2 \theta = 1
$$

$$
\tan \theta = \frac{1}{\cos^2 \theta} = \frac{1}{\sin^2 \theta}
$$

$$
\tan \theta + \frac{1}{\tan + \theta} = \frac{1}{\sin \theta \cdot \cos \theta}
$$
